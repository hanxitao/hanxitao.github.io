### 1、使用栈的思想实现flat函数

```javascript
function flat(arr) {
    const result = [],
        stack = [].concat(arr);
    
    while(stack.length !== 0) {
        const val = stack.pop();
        if (Array.isArray(val)) {
            stack.push(...val);
        } else {
            result.unshift(val);
        }
    }

    return result;
}
```

### 2、使用reduce控制拉平层数

```javascript
function flat(arr, num = 1) {
    return num > 0
        ? arr.reduce(
            (pre, cur) => {
                return pre.concat(
                    Array.isArray(cur) ? flat(cur, num - 1) : cur
                );
            }
        )
        : arr.slice();
}
```

### 3、使用Generator控制拉平层数

```javascript
function* flat(arr, num) {
    if (num === undefined) num = 1;

    for (const item of arr) {
        if (Array.isArray(item) && num > 0) {
            yield* flat(item, num - 1);
        } else {
            yield item;
        }
    }
}
```

### 4、在原型链上重写flat函数（while、concat）

```javascript
Array.prototype.fakeFlat = function(num = 1) {
    if (
        typeof num !== 'number'
        || (typeof num === 'number' && String(num) === 'NaN')
        || Number(num) < 0
    ) {
        return this;
    }

    let arr = this.concat();
    while (num > 0) {
        if (arr.some(x => Array.isArray(x))) {
            arr = [].concat.apply([], arr);
        } else {
            break;
        }

        num --;
    }

    return arr;
}
```

### 5、在原型链上重写flat函数（reduce、concat）

```javascript
Array.prototype.fakeFlat = function(num = 1) {
    if (
        typeof num !== 'number'
        || (typeof num === 'number' && String(num) === 'NaN')
        || Number(num) < 0
    ) {
        return this;
    }

    let arr = [].concat(this);
    return num > 0
        ? arr.reduce(
            (pre, cur) => {
                return pre.concat(Array.isArray(cur) ? cur.fakeFlat(num - 1) : cur)
            },
            []
        )
        : arr.filter((item, index) => {
            return arr.hasOwnProperty(index);
        });
}
```
